int FrCreater(kamaz& t, tatra& j)
{
	int ka = 0, ta = 0;
	ka = t.Fast_1 + t.Fast_2 + t.Fast_3;
	ta = j.Fast1 + j.Fast2 + j.Fast3;
	if (ka / 3 < ta / 3) {
		cout << "Победитель: " << j.name_tatra << " c средней скоростью=" << ta / 3 << " км/ч" << endl;
		cout << "результат: ";
		return -1;
	}
	else if (ka / 3 > ta / 3) {
		cout << "Побелитель: " << t.name_kamaz << " c средней скоростью=" << ka / 3 << " км/ч" << endl;
		cout << "результат: ";
		return +1;
	}
	else {
		cout << "Скорости " << t.name_kamaz << " и " << j.name_tatra << " равны: ";
		cout << ka / 3 << "==" << ta / 3 << endl;
		cout << "результат: ";
		return 0;
	}
}
