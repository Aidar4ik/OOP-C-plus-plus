#include<iostream>
using namespace std;
class Int {
	//поля класса
	long double number;
	const int max = 2147483647;
	const int min = -2147483647;
	//методы класса
public:
	Int(double n):number(n) {}  //вызываем конструктор
	
	//бинарные операторы +,-,*,/
	double operator+(const Int& x) {     //операция сложения
		long double result;
		result = this->number + x.number;
		try {
			if (result > max || result < min){
				throw "overflow";
			}
			return this->number + x.number;
		}
		catch (const char *i) {
			cout << "ошибка " << i << " - выход за границы типа int!" << endl;
			
		}
		
		
	}
	double operator-(const Int& x) {     //операция вычитания
		long double result;
		result = this->number - x.number;
		try {
			if (result > max || result < min) {
				throw "overflow";
			}
			return this->number - x.number;
		}
		catch (const char* i) {
			cout << "ошибка " << i << " - выход за границы типа int!" << endl;
		}
	}
	double operator*(const Int& x) {     //операция умножения
		long double result;
		result = this->number * x.number;
		try {
			if (result > max || result < min) {
				throw "overflow";
			}
			return this->number * x.number;
		}
		catch (const char* i) {
			cout << "ошибка " << i << " - выход за границы типа int!" << endl;
		}
	}
	double operator/(const Int& x) {     //операция деления
		long double result;
		result = this->number / x.number;
		try {
			if (result > max || result < min) {
				throw "overflow";
			}
			return this->number / x.number;
		}
		catch (const char *i) {
			cout << "ошибка " << i << " - выход за границы типа int!" << endl;
		}
	}

	//унарные операторы ++value, value++
	Int operator++() {     //префиксная форма инкремента
		++number;
		return *this;
	}
	Int operator++(int) {  //постфиксная форма инкремента
		number++;          //int это сигнал для уомпилятора, что это постфиксная версия инкремента
		return *this;
	}

	void show() {
		cout << number << endl;
	}
};
int main() {
	setlocale(LC_ALL, "rus");
	int val1, val2;
	cout << "введите значение для 'a': "; cin >> val1;
	cout << "введите значение для 'b': "; cin >> val2;
	Int a(val1), b(val2);
	cout << "Оператор '+' : "; cout << a + b << endl;
	cout << "Оператор '-' : "; cout << a - b << endl;
	cout << "Оператор '*' : "; cout << a * b << endl;
	cout << "Оператор '/' : "; cout << a / b << endl;

	cout << "Префиксная форма инкремента для 'a': "; ++a; a.show();
	cout << "Постфиксная форма инкремента для 'a': "; a++; a.show();

	system("pause>null");
	return 0;
}
