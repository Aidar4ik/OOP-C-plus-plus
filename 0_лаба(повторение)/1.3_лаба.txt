#include <iostream>
#include <conio.h>
#include<iomanip>
#include<string>
using namespace std;

struct info {
	char surname[20];
	int age;
	char specialty[20];
	double salary_size;
};

struct factory {
	info inf[100];
};

int n;
int a[100];
factory fact[100];
void menu();
void count();

int main() {
	setlocale(LC_ALL, "RUS");
	cout << "введите количество заводов N" << endl;
	cout << "Ответ пользователя: ";
	int chouse;
	while (!(cin >> chouse)) {
		cout << "Вы можете вводить только целочисленное значение. Повторите ввод :\n";
		cin.clear();
		while (cin.get() != '\n');
	}
	n = chouse;
	menu();
}

void menu() {
	system("cls");
	for (int i = 1; i <= n; i++) {
		cout << i << " - ввести данные на завод №" << i << endl;
	}
	cout << "9 - посчитать кол-во слесарей и токарей" << endl;
	cout << "0 - выход" << endl;
	cout << "Ответ пользователя: ";
	int chouse;
	while (!(cin >> chouse)) {
		cout << "Вы можете вводить только целочисленное значение. Повторите ввод :\n";
		cin.clear();
		while (cin.get() != '\n');
	}
	
	if (chouse == 0)
		exit(1);
	else if (chouse == 9) {
		count(); 
		_getch(); menu();
	}

	cout << "введите фамилию: ";
	cin >> fact[chouse].inf[a[chouse]].surname;
	cout << "введите возраст: ";
	cin >> fact[chouse].inf[a[chouse]].age;
	cout << "введите специальность: ";
	cin >> fact[chouse].inf[a[chouse]].specialty;
	cout << "введите размер зарплаты: ";
	cin >> fact[chouse].inf[a[chouse]].salary_size;
	int temp = a[chouse];
	a[chouse] = temp + 1;
	_getch(); menu();
}

void count() {
	int tokar = 0;
	int slesar = 0;
	string t = "tokar";
	string s = "slesar";
	cout << "\nфамилия\t\t" << "возраст\t" << "специальность\t" << "зарплата\t" << endl;
	for (int i = 1; i <=n;i++) {
		cout << endl;
		cout << "Завод №" << i << endl;
		cout << "________________________________________________" << endl;
		for (int j = 0; j < a[i]; j++) {
			cout << setiosflags(ios::left);
			cout << setw(16) << fact[i].inf[j].surname << fact[i].inf[j].age << "\t" << fact[i].inf[j].specialty << "\t\t" << fact[i].inf[j].salary_size << "\t" << endl;
			if (fact[i].inf[j].specialty == t) {
				tokar++;
			}
			else if (fact[i].inf[j].specialty == s) {
				slesar++;
			}
		}
	}
	cout << endl << endl;
	cout << "Количество токарей = " << tokar << endl;
	cout << "Количество слесарей = " << slesar << endl;

}
